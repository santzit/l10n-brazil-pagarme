```python
from odoo import _, api, fields, models
from odoo.exceptions import UserError, ValidationError

class PaymentProvider(models.Model):
    _inherit = 'payment.provider'
    
    # Required: Add payment provider code to selection
    code = fields.Selection(
        selection_add=[('pagarme', 'Pagarme')], 
        ondelete={'pagarme': 'set default'}
    )
    
    # Override compute methods for provider-specific behavior
    @api.depends('code')
    def _compute_view_configuration_fields(self):
        """Hide credentials page for test providers."""
        super()._compute_view_configuration_fields()
        self.filtered(lambda p: p.code == 'pagarme').show_credentials_page = False
    
    def _compute_feature_support_fields(self):
        """Define supported payment features."""
        super()._compute_feature_support_fields()
        self.filtered(lambda p: p.code == 'pagarme').update({
            'support_fees': True,
            'support_manual_capture': True,
            'support_refund': 'partial',  # Brazilian payment specifics
            'support_tokenization': True,
        })
    
    # Add constraints for Brazilian payment requirements
    @api.constrains('state', 'code')
    def _check_provider_state(self):
        if self.filtered(lambda p: p.code == 'pagarme' and p.state not in ('test', 'disabled')):
            raise UserError(_("Pagarme providers should never be enabled."))
```
